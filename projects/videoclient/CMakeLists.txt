# Target prerequisites.

	set(win32_gui_type WIN32)

# Target

	${PROJECT_NAME}
	${win32_gui_type}
	${headers}
	${sources}
	${qheaders}
	${qheaders_wrap}
	${qforms_wrap}
	${qresources_wrap}
)

	${PROJECT_NAME}
	Qt5::Core
	Qt5::Gui
	Qt5::Widgets
	Qt5::Network
	Qt5::Multimedia
	Qt5::MultimediaWidgets
	Qt5::OpenGL
	Qt5::Concurrent

	${opus_LIBRARIES}
	${humblelogging_LIBRARIES}
	qtasync
	qcorlib
	medlib
	videolib
)

)

	TARGET ${PROJECT_NAME}
	POST_BUILD
	COMMAND ${CMAKE_COMMAND} -E copy ${PROJECT_SOURCE_DIR}/res/default.ini ${PROJECT_BINARY_DIR}/$<CONFIGURATION>
	COMMAND ${CMAKE_COMMAND} -E copy ${PROJECT_SOURCE_DIR}/res/logging.conf ${PROJECT_BINARY_DIR}/$<CONFIGURATION>
